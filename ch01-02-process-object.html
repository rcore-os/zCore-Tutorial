        handle_value: HandleValue,
        desired_rights: Rights,

            .inner

            .object
            .downcast_arc::<T>()
            .map_err(|_| ZxError::WRONG_TYPE)?;

            return Err(ZxError::ACCESS_DENIED);
        }
        Ok(object)
    }
}
